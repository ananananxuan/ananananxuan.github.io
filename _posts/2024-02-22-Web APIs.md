阅读用户协议时间按钮设置
</body>
<button class="btn" disabled>我已经阅读用户协议(5)</button>
<script>
  // setInterval(函数，间隔时间)
  const button = document.querySelector('.btn')
  let i =5
  let n = setInterval(function(){
    i--
    button.innerHTML = `我已经阅读用户协议(${i})`
    if(i===0){
      clearInterval(n)
      button.disabled = false
    }
  },1000)
  
</script>
